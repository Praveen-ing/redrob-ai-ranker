% Slide 4: The Problem Statement
\begin{frame}{The Problem Statement}
  \framesubtitle{The Limitations of Traditional ATS}
  \begin{itemize}
    \item \textbf{Keyword Reliance:} Traditional systems count how many times a term appears.
    \item \textbf{Lack of Context:} ``Used Python in college'' vs.\ ``Architected a Python microservice'' score identically.
    \item \textbf{Recruiter Fatigue:} Teams waste hours manually sifting through mis-ranked profiles.
    \item \textbf{Missed Talent:} Exceptional engineers who write concise resumes are buried beneath keyword-stuffed profiles.
    \item \textbf{Ignored Signals:} Platform signals like assessment scores, notice periods, and recruiter saves go completely unused.
  \end{itemize}
\end{frame}

% Slide 5: Our Solution Overview
\begin{frame}{Our Solution Overview}
  \begin{itemize}
    \item \textbf{Semantic Understanding:} Skills grouped by meaning, not exact string matches.
    \item \textbf{Career Shape Analysis:} Trajectory of titles analyzed using the Needleman-Wunsch algorithm.
    \item \textbf{Full Signal Utilization:} Every field in the schema is used -- education tier, assessment scores, notice period, recruiter saves, offer acceptance rate.
    \item \textbf{Intent Scoring:} Measures how actively and reliably the candidate is looking.
    \item \textbf{Anti-Gaming Integrity:} Statistically detects and penalizes artificial profile inflation.
  \end{itemize}
\end{frame}

% Slide 6: The 4 Pillars
\begin{frame}{The 4 Pillars of Intelligent Ranking}
  Our engine evaluates candidates across four weighted axes:
  \begin{enumerate}
    \item \textbf{Semantic Competency (40\%):} Technical fit across clustered skill concepts.
    \item \textbf{Career Trajectory (30\%):} Pace of promotion, company pedigree, and education tier.
    \item \textbf{Behavioral Intent (20\%):} Engagement signals and readiness to join.
    \item \textbf{Location Intelligence (10\%):} Geographical alignment with the role.
  \end{enumerate}
  \vspace{1em}
  \begin{center}
    \texttt{Final Score = (Skills $\times$ 0.40) + (Experience $\times$ 0.30) + (Behavior $\times$ 0.20) + (Location $\times$ 0.10)}
  \end{center}
\end{frame}

\begin{frame}{Pillar 1: Semantic Competency (40\%)}
  \framesubtitle{Moving Beyond Exact String Matches}
  \begin{itemize}
    \item \textbf{IDF-Weighted Cluster Matching:} Rare skills carry more weight than common ones.
    \item \textbf{Endorsement Bonus:} Skills with high peer endorsements receive up to a 15\% IDF weight increase.

    \item \textbf{Cohesion Check:} Advanced skills without foundational programming knowledge incur a 20\% discount.
  \end{itemize}
\end{frame}

% Slide 10: Needleman-Wunsch
\begin{frame}{Dynamic Career Sequence Alignment}
  \framesubtitle{Applying Bioinformatics to Career Paths}
  \begin{itemize}
    \item \textbf{Algorithm:} Adapts the Needleman-Wunsch sequence alignment algorithm from genomics.
    \item \textbf{Ideal Sequence:} [Junior $\rightarrow$ Mid $\rightarrow$ Senior $\rightarrow$ Lead $\rightarrow$ Principal]
    \item \textbf{Execution:} Aligns the candidate's chronological job titles against this ideal progression.
    \item \textbf{Scoring:}
    \begin{itemize}
      \item Match Reward: +1.0
      \item Gap Penalty: -0.5 (stagnation)
      \item Mismatch Penalty: proportional to level distance
    \end{itemize}
  \end{itemize}
\end{frame}

% Slide 11: PageRank
\begin{frame}{Corporate Pedigree via PageRank}
  \begin{itemize}
    \item \textbf{The Concept:} Not all work experience is equal. Selective companies produce stronger signals.
    \item \textbf{Implementation:} A directed transition graph ($Company_A \rightarrow Company_B$) is built across the entire 100,000-candidate pool.
    \item \textbf{PageRank Execution:} Authority of each organization is determined dynamically -- no hardcoded lists.
    \item \textbf{Result:} Candidates from high-authority nodes receive a strategic, data-driven score boost.
  \end{itemize}
\end{frame}

  \framesubtitle{Geographical Alignment}

% Slide 14: Anti-Gaming
\begin{frame}{System Integrity \& Anti-Gaming}
  \framesubtitle{Defeating the Keyword Stuffers}
  \begin{itemize}
    \item \textbf{Impossible Skill Claims:} ``Expert'' proficiency + 0 months usage = instant disqualification.
    \item \textbf{Assessment Contradiction:} Claims ``Expert Python'' but scored 15/100 on the platform Python test = flagged.
    \item \textbf{Timeline Inconsistencies:} 10+ years stated but career history shows under 1 year = flagged.
    \item Flagged profiles are assigned a score of 0 and excluded from the final ranking.
  \end{itemize}
\end{frame}

% Slide 15: Shannon Entropy
\begin{frame}{Shannon Entropy Anomaly Detection}
  \framesubtitle{Statistical Profiling of Unnatural Resumes}
  \begin{itemize}
    \item \textbf{The Theory:} Natural human language has a predictable vocabulary distribution.
    \item \textbf{The Math:} Shannon Entropy is calculated over the resume text.
    $$H(X) = -\sum P(x_i) \log_2 P(x_i)$$
    \item \textbf{The Trigger:} Profiles deviating more than $2\sigma$ from the population mean are flagged automatically.
    \item \textbf{The Result:} Successfully catches machine-generated, keyword-stuffed, and white-fonted resumes.
  \end{itemize}
\end{frame}
